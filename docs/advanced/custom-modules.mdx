Learn how to create and integrate custom Terraform modules with DevPlatform CLI.

## Module Structure

DevPlatform CLI uses a modular Terraform architecture:

```
.devplatform/
└── terraform/
    ├── modules/
    │   ├── aws/
    │   │   ├── network/
    │   │   ├── database/
    │   │   └── eks-tenant/
    │   ├── azure/
    │   │   ├── network/
    │   │   ├── database/
    │   │   └── aks-tenant/
    │   └── custom/          # Your custom modules
    │       └── my-module/
    └── environments/
        └── prod/
            └── main.tf
```

## Creating a Custom Module

<Steps>
  <Step title="Create Module Directory">
    ```bash
    mkdir -p .devplatform/terraform/modules/custom/redis
    cd .devplatform/terraform/modules/custom/redis
    ```
  </Step>
  
  <Step title="Define Module Files">
    <Tabs>
      <Tab title="main.tf">
        ```hcl
        # main.tf
        resource "aws_elasticache_cluster" "redis" {
          cluster_id           = var.cluster_id
          engine               = "redis"
          node_type            = var.node_type
          num_cache_nodes      = var.num_nodes
          parameter_group_name = var.parameter_group_name
          engine_version       = var.engine_version
          port                 = 6379
          
          subnet_group_name    = aws_elasticache_subnet_group.redis.name
          security_group_ids   = [aws_security_group.redis.id]
          
          tags = merge(
            var.tags,
            {
              Name = var.cluster_id
            }
          )
        }
        
        resource "aws_elasticache_subnet_group" "redis" {
          name       = "${var.cluster_id}-subnet-group"
          subnet_ids = var.subnet_ids
        }
        
        resource "aws_security_group" "redis" {
          name        = "${var.cluster_id}-sg"
          description = "Security group for Redis cluster"
          vpc_id      = var.vpc_id
          
          ingress {
            from_port       = 6379
            to_port         = 6379
            protocol        = "tcp"
            security_groups = var.allowed_security_groups
          }
          
          egress {
            from_port   = 0
            to_port     = 0
            protocol    = "-1"
            cidr_blocks = ["0.0.0.0/0"]
          }
          
          tags = var.tags
        }
        ```
      </Tab>
      
      <Tab title="variables.tf">
        ```hcl
        # variables.tf
        variable "cluster_id" {
          description = "Redis cluster identifier"
          type        = string
        }
        
        variable "node_type" {
          description = "Cache node instance type"
          type        = string
          default     = "cache.t3.micro"
        }
        
        variable "num_nodes" {
          description = "Number of cache nodes"
          type        = number
          default     = 1
        }
        
        variable "engine_version" {
          description = "Redis engine version"
          type        = string
          default     = "7.0"
        }
        
        variable "parameter_group_name" {
          description = "Parameter group name"
          type        = string
          default     = "default.redis7"
        }
        
        variable "vpc_id" {
          description = "VPC ID"
          type        = string
        }
        
        variable "subnet_ids" {
          description = "Subnet IDs for Redis"
          type        = list(string)
        }
        
        variable "allowed_security_groups" {
          description = "Security groups allowed to access Redis"
          type        = list(string)
        }
        
        variable "tags" {
          description = "Tags to apply to resources"
          type        = map(string)
          default     = {}
        }
        ```
      </Tab>
      
      <Tab title="outputs.tf">
        ```hcl
        # outputs.tf
        output "endpoint" {
          description = "Redis endpoint"
          value       = aws_elasticache_cluster.redis.cache_nodes[0].address
        }
        
        output "port" {
          description = "Redis port"
          value       = aws_elasticache_cluster.redis.cache_nodes[0].port
        }
        
        output "connection_string" {
          description = "Redis connection string"
          value       = "redis://${aws_elasticache_cluster.redis.cache_nodes[0].address}:${aws_elasticache_cluster.redis.cache_nodes[0].port}"
          sensitive   = true
        }
        
        output "security_group_id" {
          description = "Security group ID"
          value       = aws_security_group.redis.id
        }
        ```
      </Tab>
    </Tabs>
  </Step>
  
  <Step title="Use Module in Environment">
    ```hcl
    # .devplatform/terraform/environments/prod/main.tf
    module "redis" {
      source = "../../modules/custom/redis"
      
      cluster_id                = "myapp-prod-redis"
      node_type                 = "cache.r6g.large"
      num_nodes                 = 2
      engine_version            = "7.0"
      
      vpc_id                    = module.network.vpc_id
      subnet_ids                = module.network.private_subnet_ids
      allowed_security_groups   = [module.eks_tenant.security_group_id]
      
      tags = local.common_tags
    }
    
    # Output for use in Kubernetes
    output "redis_endpoint" {
      value     = module.redis.endpoint
      sensitive = true
    }
    ```
  </Step>
</Steps>

## Module Best Practices

<AccordionGroup>
  <Accordion title="Use Variables for Flexibility">
    ```hcl
    variable "environment" {
      description = "Environment name"
      type        = string
      validation {
        condition     = contains(["dev", "staging", "prod"], var.environment)
        error_message = "Environment must be dev, staging, or prod"
      }
    }
    
    variable "instance_type" {
      description = "Instance type"
      type        = string
      default     = "cache.t3.micro"
    }
    
    locals {
      # Environment-specific defaults
      instance_type = var.environment == "prod" ? "cache.r6g.large" : var.instance_type
    }
    ```
  </Accordion>
  
  <Accordion title="Add Comprehensive Outputs">
    ```hcl
    output "endpoint" {
      description = "Service endpoint"
      value       = aws_elasticache_cluster.redis.cache_nodes[0].address
    }
    
    output "port" {
      description = "Service port"
      value       = aws_elasticache_cluster.redis.cache_nodes[0].port
    }
    
    output "arn" {
      description = "Resource ARN"
      value       = aws_elasticache_cluster.redis.arn
    }
    
    output "security_group_id" {
      description = "Security group ID for ingress rules"
      value       = aws_security_group.redis.id
    }
    ```
  </Accordion>
  
  <Accordion title="Use Consistent Tagging">
    ```hcl
    locals {
      common_tags = merge(
        var.tags,
        {
          ManagedBy   = "devplatform-cli"
          Environment = var.environment
          Application = var.application_name
          Module      = "custom/redis"
        }
      )
    }
    
    resource "aws_elasticache_cluster" "redis" {
      # ...
      tags = local.common_tags
    }
    ```
  </Accordion>
  
  <Accordion title="Add Validation">
    ```hcl
    variable "num_nodes" {
      description = "Number of cache nodes"
      type        = number
      default     = 1
      
      validation {
        condition     = var.num_nodes >= 1 && var.num_nodes <= 6
        error_message = "Number of nodes must be between 1 and 6"
      }
    }
    ```
  </Accordion>
</AccordionGroup>

## Integrating with Kubernetes

Pass Terraform outputs to Kubernetes:

<Tabs>
  <Tab title="Create Kubernetes Secret">
    ```hcl
    # Create secret with Redis connection info
    resource "kubernetes_secret" "redis" {
      metadata {
        name      = "redis-connection"
        namespace = var.namespace
      }
      
      data = {
        host     = module.redis.endpoint
        port     = module.redis.port
        url      = module.redis.connection_string
      }
      
      type = "Opaque"
    }
    ```
  </Tab>
  
  <Tab title="Use in Deployment">
    ```yaml
    apiVersion: apps/v1
    kind: Deployment
    metadata:
      name: myapp
    spec:
      template:
        spec:
          containers:
          - name: app
            image: myapp:latest
            env:
            - name: REDIS_HOST
              valueFrom:
                secretKeyRef:
                  name: redis-connection
                  key: host
            - name: REDIS_PORT
              valueFrom:
                secretKeyRef:
                  name: redis-connection
                  key: port
    ```
  </Tab>
</Tabs>

## Testing Custom Modules

<Steps>
  <Step title="Validate Syntax">
    ```bash
    cd .devplatform/terraform/modules/custom/redis
    terraform validate
    ```
  </Step>
  
  <Step title="Format Code">
    ```bash
    terraform fmt -recursive
    ```
  </Step>
  
  <Step title="Test in Dev Environment">
    ```bash
    devplatform create --app myapp --env dev
    ```
  </Step>
  
  <Step title="Verify Resources">
    ```bash
    # Check Terraform state
    cd .devplatform/terraform/myapp/dev
    terraform state list
    
    # Verify outputs
    terraform output redis_endpoint
    ```
  </Step>
</Steps>

## Example: Custom Monitoring Module

```hcl
# modules/custom/monitoring/main.tf
resource "aws_cloudwatch_dashboard" "main" {
  dashboard_name = var.dashboard_name
  
  dashboard_body = jsonencode({
    widgets = [
      {
        type = "metric"
        properties = {
          metrics = [
            ["AWS/RDS", "CPUUtilization", { stat = "Average" }],
            ["AWS/RDS", "DatabaseConnections", { stat = "Sum" }]
          ]
          period = 300
          stat   = "Average"
          region = var.region
          title  = "Database Metrics"
        }
      },
      {
        type = "metric"
        properties = {
          metrics = [
            ["AWS/EKS", "cluster_failed_node_count", { stat = "Average" }],
            ["AWS/EKS", "cluster_node_count", { stat = "Average" }]
          ]
          period = 300
          stat   = "Average"
          region = var.region
          title  = "Kubernetes Metrics"
        }
      }
    ]
  })
}

resource "aws_cloudwatch_metric_alarm" "high_cpu" {
  alarm_name          = "${var.application_name}-high-cpu"
  comparison_operator = "GreaterThanThreshold"
  evaluation_periods  = "2"
  metric_name         = "CPUUtilization"
  namespace           = "AWS/RDS"
  period              = "300"
  statistic           = "Average"
  threshold           = "80"
  alarm_description   = "This metric monitors RDS CPU utilization"
  alarm_actions       = [var.sns_topic_arn]
}
```
